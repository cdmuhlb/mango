53 int main(int argc, char *argv[]) {
54  int ierr;
55 
56  if (verbose_level > 0) std::cout << "Hello world from chwirut_c.\n";
57 
58  ierr = MPI_Init(&argc, &argv);
59  if (ierr != 0) {
60  std::cout << "\nError in MPI_Init.\n";
61  exit(1);
62  }

64  datapoint yt_data[N_terms];
65 
66  init_data(yt_data);
67 
68  double state_vector[3] = {0.15, 0.008, 0.01};
69  double targets[N_terms];
70  double sigmas[N_terms];
71  double best_residual_function[N_terms];
72  for (int j=0; j<N_terms; j++) {
73  targets[j] = 0.0;
74  sigmas[j] = 1.0;
75  }
76  mango::Least_squares_problem myprob(3, state_vector, N_terms, targets, sigmas, best_residual_function, &residual_function, argc, argv);
77 
78  std::string extension = "chwirut_c";
79  // myprob.set_algorithm(mango::PETSC_POUNDERS);
80  // myprob.set_algorithm("nlopt_ln_neldermead");
81  myprob.set_verbose(verbose_level);
82  myprob.read_input_file("../input/mango_in." + extension);
83  myprob.set_output_filename("../output/mango_out." + extension);
84  myprob.mpi_init(MPI_COMM_WORLD);
85  myprob.mpi_partition.write("../output/mango_mpi." + extension);
86  // myprob.centered_differences = true;
87  myprob.set_max_function_evaluations(2000);
88  myprob.set_print_residuals_in_output_file(false);
89  myprob.set_user_data((void*)yt_data); // This passes the (y,t) data to the residual function
90 
91  double best_objective_function;
92  if (myprob.mpi_partition.get_proc0_worker_groups()) {
93  best_objective_function = myprob.optimize();
94  myprob.mpi_partition.stop_workers();
95  } else {
96  worker(&myprob, yt_data);
97  }
98 
99  if (myprob.mpi_partition.get_proc0_world() && (verbose_level > 0)) {
100  std::cout << "Best state vector: " << std::setprecision(16);
101  for (int j=0; j<3; j++) std::cout << state_vector[j] << " ";
102  std::cout << "\nBest objective function: " << best_objective_function << "\nFirst 3 elements of best residual vector: ";
103  for (int j=0; j<3; j++) std::cout << best_residual_function[j] << " ";
104  std::cout << "\nBest function evaluation was " << myprob.get_best_function_evaluation() << "\n";
105  }
106 
107  MPI_Finalize();
108 
109  return 0;
110 }

115 void do_work(int N_parameters, double* x, double* f, int start_index, int stop_index, datapoint* data) {
116  for (int j=start_index; j <= stop_index; j++) {
117  f[j] = data[j].y - std::exp(-x[0] * data[j].t) / (x[1] + x[2] * data[j].t);
118  }
119 }

124 void residual_function(int* N_parameters, const double* x, int* N_terms_copy, double* f, int* failed, mango::Problem* this_problem, void* void_user_data) {
125  int j, start_index, stop_index;
126  if (verbose_level > 0) std::cout << "C residual function called with N="<< *N_parameters << "\n";
127 
128  datapoint* yt_data = (datapoint*)void_user_data;
129 
130  MPI_Status mpi_status;
131  MPI_Comm comm_worker_groups = this_problem->mpi_partition.get_comm_worker_groups();
132  int N_procs_worker_groups = this_problem->mpi_partition.get_N_procs_worker_groups();
133 
134  // Mobilize the workers in the group with this group leader:
135  this_problem->mpi_partition.mobilize_workers();
136 
137  // Send the state vector to all the other procs:
138  int N = *N_parameters;
139  MPI_Bcast((double*)x, N, MPI_DOUBLE, 0, comm_worker_groups);
140 
141  // Compute the residual terms on this proc, or receive the terms from the worker procs.
142  for (j=0; j < N_procs_worker_groups; j++) {
143  partition_work(j, N_procs_worker_groups, &start_index, &stop_index);
144  if (j==0) {
145  do_work(*N_parameters, (double*)x, f, start_index, stop_index, yt_data);
146  } else {
147  MPI_Recv(&f[start_index], stop_index - start_index + 1, MPI_DOUBLE, j, j, comm_worker_groups, &mpi_status);
148  }
149  }
150 
151  *failed = false;
152 }

157 void worker(mango::Least_squares_problem* myprob, datapoint* yt_data) {
158  MPI_Comm mpi_comm_worker_groups = myprob->mpi_partition.get_comm_worker_groups();
159  int N_parameters = myprob->get_N_parameters();
160  int mpi_rank_worker_groups = myprob->mpi_partition.get_rank_worker_groups();
161 
162  int start_index, stop_index;
163  double x[3];
164  double f[N_terms];
165 
166  partition_work(mpi_rank_worker_groups, myprob->mpi_partition.get_N_procs_worker_groups(), &start_index, &stop_index);
167 
168  while (myprob->mpi_partition.continue_worker_loop()) {
169  if (verbose_level > 0) std::cout<< "Proc " << std::setw(5) << myprob->mpi_partition.get_rank_world() << " is processing indices " << start_index << " to " << stop_index << "\n";
170 
171  // Get the state vector
172  MPI_Bcast(x, N_parameters, MPI_DOUBLE, 0, mpi_comm_worker_groups);
173 
174  do_work(N_parameters, x, f, start_index, stop_index, yt_data);
175 
176  // Send my terms of the residual back to the master proc.
177  MPI_Send(&f[start_index], stop_index - start_index + 1, MPI_DOUBLE, 0, mpi_rank_worker_groups, mpi_comm_worker_groups);
178  }
179  if (verbose_level > 0) std::cout << "Proc " << std::setw(5) << myprob->mpi_partition.get_rank_world() << " is exiting.\n";
180 }
