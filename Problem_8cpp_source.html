88 #define bold_line "****************************************************************************************"
89 void mango::Problem::mpi_init(MPI_Comm mpi_comm_world) {
90  // This method basically just calls MPI_Partition::init, but first checks to see if the algorithm
91  // chosen can support parallel function evaluations. If not, N_worker_groups is set to 1.
92 
93  if (solver->algorithm < 0) throw std::runtime_error("Error in mango::Problem::mpi_init. Algorithm cannot be negative.");
94  if (solver->algorithm >= NUM_ALGORITHMS) throw std::runtime_error("Error in mango::Problem::mpi_init. Algorithm is too large.");
95 
96  mpi_partition.verbose = solver->verbose;
97 
98  if (algorithms[solver->algorithm].parallel) {
99  int mpi_rank_world, N_procs_world;
100  MPI_Comm_size(mpi_comm_world, &N_procs_world);
101  MPI_Comm_rank(mpi_comm_world, &mpi_rank_world);
102 
103  if ((N_procs_world > 1) && (mpi_partition.get_N_worker_groups() == 1) && (mpi_rank_world==0)) {
104  std::cerr << bold_line << std::endl;
105  std::cerr << "WARNING!!! You have chosen an algorithm that can exploit concurrent function evaluations" << std::endl;
106  std::cerr << "but you have set N_worker_groups=1. You probably want a larger value." << std::endl;
107  std::cerr << bold_line << std::endl;
108  }
109  } else {
110  // There is no point having >1 worker groups with these algorithms.
111  mpi_partition.set_N_worker_groups(1);
112  }
113 
114  mpi_partition.init(mpi_comm_world);
115 }

124 void mango::Problem::set_relative_bound_constraints(double min_factor, double max_factor, double min_radius, bool preserve_sign) {
125  if (min_factor < 0) throw std::runtime_error("mango::Problem::set_relative_bound_constraints: min_factor must be >= 0.");
126  if (min_factor > 1) throw std::runtime_error("mango::Problem::set_relative_bound_constraints: min_factor must be <= 1.");
127 
128  if (max_factor < 1) throw std::runtime_error("mango::Problem::set_relative_bound_constraints: max_factor must be >= 1.");
129 
130  if (min_radius < 0) throw std::runtime_error("mango::Problem::set_relative_bound_constraints: min_radius must be >= 0.");
131 
132  if (! solver->bound_constraints_set) throw std::runtime_error("mango::Problem::set_relative_bound_constraints can only be called after bound constraints are set.");
133 
134  int j;
135  double temp;
136  if (preserve_sign) {
137  // Approach in which the sign of the variable is preserved.
138  for (j=0; j < solver->N_parameters; j++) {
139  if (solver->state_vector[j] > 0) {
140  // Initial value and both bounds are positive
141  solver->lower_bounds[j] = min_factor * solver->state_vector[j];
142  solver->upper_bounds[j] = max_factor * solver->state_vector[j];
143  if (solver->upper_bounds[j] - solver->state_vector[j] < min_radius) solver->upper_bounds[j] = solver->state_vector[j] + min_radius;
144  if (solver->state_vector[j] - solver->lower_bounds[j] < min_radius) solver->lower_bounds[j] = solver->state_vector[j] - min_radius;
145  if (solver->lower_bounds[j] < 0) solver->lower_bounds[j] = 0;
146  } else if (solver->state_vector[j] < 0) {
147  // Initial value and both bounds are negative
148  solver->lower_bounds[j] = max_factor * solver->state_vector[j];
149  solver->upper_bounds[j] = min_factor * solver->state_vector[j];
150  if (solver->upper_bounds[j] - solver->state_vector[j] < min_radius) solver->upper_bounds[j] = solver->state_vector[j] + min_radius;
151  if (solver->state_vector[j] - solver->lower_bounds[j] < min_radius) solver->lower_bounds[j] = solver->state_vector[j] - min_radius;
152  if (solver->upper_bounds[j] > 0) solver->upper_bounds[j] = 0;
153  } else {
154  // Initial value is 0, so there is a bound of each sign
155  assert(solver->state_vector[j] == 0);
156  solver->lower_bounds[j] = -min_radius;
157  solver->upper_bounds[j] = min_radius;
158  }
159  if (! (solver->upper_bounds[j] >= solver->lower_bounds[j])) std::cout << "ub:" << solver->upper_bounds[j] << " lb:" << solver->lower_bounds[j] << "\n";
160  assert(solver->upper_bounds[j] >= solver->lower_bounds[j]);
161  }
162  } else {
163  // Approach in which the sign of the variable is not preserved.
164  for (j=0; j < solver->N_parameters; j++) {
165  temp = max_factor * abs(solver->state_vector[j]);
166  if (temp < min_radius) temp = min_radius;
167  solver->lower_bounds[j] = -temp;
168  solver->upper_bounds[j] = temp;
169  assert(solver->upper_bounds[j] >= solver->lower_bounds[j]);
170  }
171  }
172 }
