21 ! The standard 2D Rosenbrock banana function:
22 ! f(x,y) = (x - 1)^2 + 100 * (y - x^2)^2
23 ! The optimum is at (x,y) = (1,1), and the objective function there is f=0.
24 !
25 ! This example demonstrates several things:
26 ! * Unconstrained least-squares minimization.
27 ! * The problem is more challenging than the 'quadratic_c' example due to the objective function having a steep curved valley.
28 ! * Passing a double to the objective/residual function through the 'user_data'.
29 
30 #define verbose_level 0
31 
32 program rosenbrock
33 
34  use mango
35  use iso_c_binding, only: c_loc
36 
37  implicit none
38 
39  include 'mpif.h'
40 
41  character(len=*), parameter :: extension = "rosenbrock_f"
42  integer :: ierr, n_procs, mpi_rank
43  logical :: proc0
44  type(mango_problem) :: problem
45  double precision, dimension(2) :: state_vector = (/ 0.0d+0, 0.0d+0 /)
46  double precision, dimension(2) :: targets = (/ 1.0d+0, 0.0d+0 /)
47  double precision, dimension(2) :: sigmas = (/ 1.0d+0, 0.1d+0 /)
48  double precision :: best_residual_function(2), best_objective_function
49 
50  double precision, target :: my_data = 2.71828182845905d+0 ! "target" attribute is necessary since we will use my_data as an argument of c_loc.
51 
52  !---------------------------------------------
53 
54  if (verbose_level > 0) print *,"Hello world from rosenbrock_f"
55  call mpi_init(ierr)
56 
57  !call mango_problem_create(problem,2,state_vector,dummy,objective_function)
58  call mango_problem_create_least_squares(problem, 2, state_vector, 2, targets, sigmas, best_residual_function, residual_function)
59  if (verbose_level > 0) print *,"Here comes state vector:",state_vector
60  !call mango_set_algorithm(problem, 2)
61  !call mango_set_algorithm_from_string(problem, "nlopt_ln_praxis")
62  call mango_set_verbose(problem, verbose_level)
63  call mango_read_input_file(problem, "../input/mango_in." // extension)
64  call mango_set_output_filename(problem, "../output/mango_out." // extension)
65  call mango_mpi_init(problem, mpi_comm_world)
66  call mango_mpi_partition_write(problem, "../output/mango_mpi." // extension)
67  call mango_set_max_function_evaluations(problem, 2000)
68 
69  call mango_set_user_data(problem, c_loc(my_data))
70 
71  if (mango_get_proc0_worker_groups(problem)) then
72  best_objective_function = mango_optimize(problem)
73  call mango_stop_workers(problem)
74  else
75  call worker(problem)
76  end if
77 
78  if (mango_get_proc0_world(problem) .and. (verbose_level > 0)) then
79  print *,"Best state vector:",state_vector
80  print *,"Best objective function: ",best_objective_function
81  print *,"Best residual function: ",best_residual_function
82  print *,"Best function evaluation was ",mango_get_best_function_evaluation(problem)
83  end if
84 
85  call mango_problem_destroy(problem)
86 
87  call mpi_finalize(ierr)
88 
89  if (verbose_level > 0) print *,"Good bye!"

93 subroutine residual_function(N_parameters, x, N_terms, f, failed, problem, void_user_data)
94  use iso_c_binding
95  implicit none
96  integer(C_int), intent(in) :: N_parameters
97  real(C_double), intent(in) :: x(N_parameters)
98  integer(C_int), intent(in) :: N_terms
99  real(C_double), intent(out) :: f(N_terms)
100  integer(C_int), intent(out) :: failed
101  type(mango_problem), value, intent(in) :: problem
102  type(c_ptr), value, intent(in) :: void_user_data
103 
104  double precision, pointer :: user_data
105 
106  if (verbose_level > 0) print *,"Hi from fortran. N_parameters=",n_parameters," size(x)=",size(x)
107 
108  call mango_mobilize_workers(problem)
109 
110  ! Check that user data was passed correctly
111  call c_f_pointer(void_user_data, user_data) ! This line effectively casts (void*) to (double*)
112  if (abs(user_data - 2.71828182845905d+0) > 1d-13) then
113  print *,"Error passing user_data to the residual function. user_data=",user_data
114  stop
115  end if
116 
117  !f = (x(1) - 1) * (x(1) - 1) + 100 * (x(2) - x(1)*x(1)) * (x(2) - x(1)*x(1))
118  f(1) = x(1)
119  f(2) = x(2) - x(1) * x(1)
120  failed = 0
121 
122 end subroutine residual_function
123 
124 end program rosenbrock
