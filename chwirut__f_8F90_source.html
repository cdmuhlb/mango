26 program chwirut

39  integer, parameter :: n_terms = 214
40  integer, parameter :: n_parameters = 3
41  double precision :: y(n_terms), t(n_terms)
42  double precision, dimension(N_parameters) :: state_vector = (/ 0.15d+0, 0.008d+0, 0.01d+0 /)

56  !call mango_set_algorithm(problem, 2)

91 subroutine do_work(N_parameters, x, N_terms, f, start_index, stop_index)
92 
93  integer, intent(in) :: N_parameters, N_terms, start_index, stop_index
94  double precision, intent(in) :: x(N_parameters)
95  double precision, intent(out) :: f(N_terms)
96  integer :: j
97 
98  do j = start_index, stop_index
99  f(j) = y(j) - exp(-x(1)*t(j))/(x(2)+x(3)*t(j))
100  end do

107 subroutine residual_function(N_parameters, x, N_terms, f, failed, problem, user_data)
108  use iso_c_binding
109  implicit none
110  integer(C_int), intent(in) :: N_parameters
111  real(C_double), intent(in) :: x(N_parameters)
112  integer(C_int), intent(in) :: N_terms
113  real(C_double), intent(out) :: f(N_terms)
114  integer(C_int), intent(out) :: failed
115  type(mango_problem), value, intent(in) :: problem
116  type(c_ptr), value, intent(in) :: user_data
117  integer :: ierr, j, start_index, stop_index, N_procs_worker_groups
118  integer :: mpi_status(MPI_STATUS_SIZE)
119  integer :: mpi_comm_worker_groups
120 
121  if (verbose_level > 0) print *,"Hi from fortran. N_parameters=",n_parameters," size(x)=",size(x)
122  n_procs_worker_groups = mango_get_n_procs_worker_groups(problem)
123  mpi_comm_worker_groups = mango_get_mpi_comm_worker_groups(problem)
124 
125  ! Mobilize the workers in the group with this group leader:
126  call mango_mobilize_workers(problem)
127 
128  ! Send the state vector to all the other procs:
129  call mpi_bcast(x, n_parameters, mpi_double, 0, mpi_comm_worker_groups, ierr)
130 
131  ! Compute the residual terms on this proc, or receive the terms from the worker procs.
132  do j = 0, (n_procs_worker_groups-1)
133  call partition_work(n_terms, j, n_procs_worker_groups, start_index, stop_index)
134  !print "(4(a,i4))","residual function, j=",j," of",N_procs_worker_groups,", start_index=",start_index,", stop_index=",stop_index
135  if (j==0) then
136  call do_work(n_parameters, x, n_terms, f, start_index, stop_index)
137  else
138  call mpi_recv(f(start_index:stop_index), stop_index - start_index + 1, mpi_double, j, j, mpi_comm_worker_groups,mpi_status,ierr)
139  end if
140  end do
141 
142  failed = 0
143 
144 end subroutine residual_function

149 subroutine worker(problem)
150 
151  use mango
152 
153  implicit none
154 
155  include 'mpif.h'
156 
157  type(mango_problem) :: problem
158  integer :: ierr
159  double precision, allocatable :: f(:), x(:)
160  integer :: start_index, stop_index, N_terms, mpi_rank_worker_groups, N_parameters
161  integer :: mpi_comm_worker_groups
162 
163  mpi_comm_worker_groups = mango_get_mpi_comm_worker_groups(problem)
164  n_terms = mango_get_n_terms(problem)
165  n_parameters = mango_get_n_parameters(problem)
166  mpi_rank_worker_groups = mango_get_mpi_rank_worker_groups(problem)
167  allocate(f(n_terms))
168  allocate(x(n_parameters))
169  call partition_work(n_terms, mpi_rank_worker_groups, mango_get_n_procs_worker_groups(problem), start_index, stop_index)
170 
171  do while (mango_continue_worker_loop(problem))
172  if (verbose_level > 0) print "(a,i4,a,i4,a,i4)", "Proc",mango_get_mpi_rank_world(problem)," is processing indices",start_index," to",stop_index
173 
174  ! Get the state vector
175  call mpi_bcast(x, n_parameters, mpi_double, 0, mpi_comm_worker_groups, ierr)
176 
177  call do_work(n_parameters, x, n_terms, f, start_index, stop_index)
178 
179  ! Send my terms of the residual back to the master proc.
180  call mpi_send(f(start_index:stop_index), stop_index - start_index + 1, mpi_double, 0,mpi_rank_worker_groups,mpi_comm_worker_groups,ierr)
181  end do
182 
183  if (verbose_level > 0) print "(a,i4,a)", "Proc",mango_get_mpi_rank_world(problem)," is exiting."
184  deallocate(f,x)
185 
186 end subroutine worker
187 
188 end program chwirut
